Algoritmo NUMEROSECRETO
	//Definiciones
	Definir num, numero_secreto, intentos_usuario Como Entero
	
	//Base
	Escribir "PC eligira un número aleatorio entre el 1 y el 100"
	Escribir "Piensa en un numero y digitalo"
	numero_secreto = Aleatorio(1 , 100)
	intentos_usuario = 0
	num = 0
	
	//Operacion
	Mientras num <> numero_secreto Hacer
		Leer num
		Si num < numero_secreto Entonces
			Escribir num, " es menor a comparación del número secreto"
		SiNo
			Escribir num, " es mayor a comparación del número secreto"
		FinSi
		intentos_usuario = intentos_usuario + 1
	FinMientras
	
	//Final
	Si num = numero_secreto Entonces
		Borrar Pantalla
		Escribir "Adivinaste el número, este era " numero_secreto
		Escribir "Intentos = ", intentos_usuario
	FinSi
	
FinAlgoritmo
